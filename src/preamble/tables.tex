\usepackage{tabularx}
\usepackage{hhline}
\usepackage{xcolor}        % background colors
\PassOptionsToPackage{table}{xcolor}
\usepackage{booktabs}      % better looking table lines
\usepackage{array}         % custom column styles
\usepackage{colortbl}      % colored cells
\usepackage{multirow}      % multiple rows

\definecolor{HeaderColor}{RGB}{230, 231, 232}

\newenvironment{table-custom}[2]
{
    \table[ht!]
    \centering
    \caption{#2}
    \setlength{\tabcolsep}{12pt}          % column spacing
    \renewcommand{\arraystretch}{1.2}     % row spacing
    \arrayrulecolor{black}                % table border color
    \tabularx{\textwidth}{#1}
    \hline
    \rowcolor{HeaderColor}
}
{
    \endtabularx
    \endtable
}
