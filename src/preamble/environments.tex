\newenvironment{enum}
{
    \begin{enumerate}
        \setlength\itemsep{0.5em}
}
{
    \end{enumerate}
}

\newenvironment{enum-alph}
{
    \begin{enumerate}[label=(\alph*)]
}
{
    \end{enumerate}
}

\newenvironment{enum-roman}
{
    \begin{enumerate}[label=(\roman*)]
}
{
    \end{enumerate}
}

\newenvironment{list}
{
    \begin{itemize}
}
{
    \end{itemize}
}

\newenvironment{problems}
{
    \subsection{Problems}
    \begin{enumerate}
}
{
    \end{enumerate}
}

\newenvironment{solutions}
{
    \subsection{Solutions}
    \begin{enumerate}
}
{
    \end{enumerate}
}

   \noindent \hangindent=\wd0 \box0 \ignorespaces
}

\def\answerT{\setbox0=\hbox{\bf Answer: }%
   \noindent \hangindent=\wd0 \box0 \ignorespaces
}

\newenvironment{sample}[1]
{
    \sampleT
    #1

    \answerT
}
